## 解法

```cpp
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        if (!head || !head->next) return head;

        // Step 1: 計算鏈表長度
        int length = 0;
        ListNode* node = head;
        while (node) {
            length++;
            node = node->next;
        }

        ListNode dummy(0);
        dummy.next = head;

        // Step 2: 自底向上歸併排序
        for (int step = 1; step < length; step *= 2) {
            ListNode* prev = &dummy;
            ListNode* curr = dummy.next;

            while (curr) {
                // 拆分兩段
                ListNode* left = curr;
                ListNode* right = split(left, step);
                curr = split(right, step);

                // 合併兩段
                prev->next = merge(left, right);

                // 移動 prev 到合併後的尾端
                while (prev->next) prev = prev->next;
            }
        }

        return dummy.next;
    }

private:
    // 拆分鏈表，返回第二段的起始節點
    ListNode* split(ListNode* head, int step) {
        for (int i = 1; head && i < step; i++) {
            head = head->next;
        }
        if (!head) return nullptr;

        ListNode* second = head->next;
        head->next = nullptr;
        return second;
    }

    // 合併兩個有序鏈表
    ListNode* merge(ListNode* l1, ListNode* l2) {
        ListNode dummy(0);
        ListNode* tail = &dummy;

        while (l1 && l2) {
            if (l1->val < l2->val) {
                tail->next = l1;
                l1 = l1->next;
            } else {
                tail->next = l2;
                l2 = l2->next;
            }
            tail = tail->next;
        }

        tail->next = l1 ? l1 : l2;
        return dummy.next;
    }
};
```
